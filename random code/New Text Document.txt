def get_last_n_dates(yyyymm, n):
    # Convert yyyymm to a date object
    year = int(yyyymm[:4])
    month = int(yyyymm[4:])
    
    # Get the last day of the month
    last_day = datetime(year, month, 1) + timedelta(days=32)  # move to next month
    last_day = datetime(last_day.year, last_day.month, 1) - timedelta(days=1)  # get the last day of the given month
    
    # Generate the list of the last n dates in yyyymm format
    dates = []
    for i in range(n):
        date = last_day - timedelta(days=i)
        dates.append(date.strftime('%Y%m'))  # Format as YYYYMM
    
    return dates
