
def extract_date_from_string(input_string):
    # Use regex to find any date in the format YYYYMM (8 digits)
    match = re.search(r'\d{6}', input_string)
    
    if match:
        date_str = match.group()
        # Convert to a date object to ensure it's valid
        try:
            date = datetime.strptime(date_str, '%Y%m')
            return date.strftime('%Y%m')
        except ValueError:
            return "Invalid date format in string"
    else:
        return "No valid date found"
