from datetime import datetime, timedelta
import calendar

def get_last_n_months(yyyymm, n):
    # Convert yyyymm to a date object
    year = int(yyyymm[:4])
    month = int(yyyymm[4:])
    
    months = []
    
    for i in range(n):
        # Calculate the month and year for the i-th previous month
        new_month = month - i
        new_year = year
        
        if new_month <= 0:
            new_month += 12
            new_year -= 1
        
        # Format the date in YYYYMM format
        months.append(f'{new_year:04d}{new_month:02d}')
    
    return months
